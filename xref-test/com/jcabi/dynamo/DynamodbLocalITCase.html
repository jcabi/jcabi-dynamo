5   package com.jcabi.dynamo;
6   
7   import java.util.Iterator;
8   import org.apache.commons.lang3.RandomStringUtils;
9   import org.hamcrest.MatcherAssert;
10  import org.hamcrest.Matchers;
11  import org.junit.jupiter.api.Assumptions;
12  import org.junit.jupiter.api.BeforeEach;
13  import org.junit.jupiter.api.Test;
14  
15  /**
16   * Integration case with DynamoDB Local.
17   *
18   * @since 0.23
19   */
20  @SuppressWarnings("PMD.AvoidDuplicateLiterals")
21  final class DynamodbLocalITCase {
22  
23      @BeforeEach
24      void itTestCheck() {
25          Assumptions.assumeFalse(
26              System.getProperty("failsafe.ddl.port", "").isEmpty()
27                  || System.getProperty("failsafe.ddl.key", "").isEmpty()
28                  || System.getProperty("failsafe.ddl.secret", "").isEmpty(),
29              "DynamoDbLocal is not up and running, that's why this test is skipped"
30          );
31      }
32  
33      @Test
34      @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
35      void worksWithAmazon() throws Exception {
36          final Table tbl = new Region.Simple(
37              new Credentials.Direct(
38                  new Credentials.Simple(
39                      System.getProperty("failsafe.ddl.key"),
40                      System.getProperty("failsafe.ddl.secret")
41                  ),
42                  Integer.parseInt(System.getProperty("failsafe.ddl.port"))
43              )
44          ).table("talks");
45          for (int idx = 0; idx < 5; ++idx) {
46              tbl.put(
47                  new Attributes()
48                      .with("room", idx)
49                      .with("title", RandomStringUtils.randomAlphanumeric(10))
50              );
51          }
52          MatcherAssert.assertThat(
53              "should has size 1",
54              tbl.frame()
55                  .where("room", Conditions.equalTo(0))
56                  .through(new QueryValve().withLimit(1)),
57              Matchers.hasSize(1)
58          );
59          MatcherAssert.assertThat(
60              "should equals to 0",
61              tbl.frame()
62                  .where("room", Conditions.equalTo(0))
63                  .through(new ScanValve())
64                  .iterator().next()
65                  .get("room")
66                  .n(),
67              Matchers.equalTo("0")
68          );
69          final Iterator<Item> items = tbl.frame().iterator();
70          items.next();
71          items.remove();
72      }
73  
74  }
