5   package com.jcabi.dynamo.retry;
6   
7   import com.jcabi.dynamo.Attributes;
8   import com.jcabi.dynamo.Table;
9   import java.io.IOException;
10  import org.hamcrest.MatcherAssert;
11  import org.hamcrest.Matchers;
12  import org.junit.jupiter.api.Test;
13  import org.mockito.Mockito;
14  
15  /**
16   * Test case for {@link ReTable}.
17   * @since 0.1
18   */
19  final class ReTableTest {
20  
21      @Test
22      void retriesDelete() throws Exception {
23          final Table table = Mockito.mock(Table.class);
24          final Attributes attrs = new Attributes();
25          final String msg = "Exception!";
26          Mockito.doThrow(new IOException(msg)).when(table)
27              .delete(attrs);
28          final Table retry = new ReTable(table);
29          try {
30              retry.delete(attrs);
31          } catch (final IOException ex) {
32              MatcherAssert.assertThat(
33                  "should be equal to 'Exception!'",
34                  ex.getMessage(),
35                  Matchers.equalTo(msg)
36              );
37          }
38          Mockito.verify(table, Mockito.times(3)).delete(attrs);
39      }
40  }
