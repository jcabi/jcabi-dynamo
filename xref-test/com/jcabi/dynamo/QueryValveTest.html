5   package com.jcabi.dynamo;

7   import com.google.common.collect.ImmutableMap;
8   import java.util.ArrayList;
9   import java.util.Collections;
10  import org.hamcrest.MatcherAssert;
11  import org.hamcrest.Matchers;
12  import org.junit.jupiter.api.Test;
13  import org.mockito.Mockito;
14  import software.amazon.awssdk.services.dynamodb.DynamoDbClient;
15  import software.amazon.awssdk.services.dynamodb.model.AttributeValue;
16  import software.amazon.awssdk.services.dynamodb.model.ConsumedCapacity;
17  import software.amazon.awssdk.services.dynamodb.model.QueryRequest;
18  import software.amazon.awssdk.services.dynamodb.model.QueryResponse;
19  
20  /**
21   * Test case for {@link QueryValve}.
22   * @since 0.1
23   */
24  final class QueryValveTest {
25  
26      @Test
27      @SuppressWarnings("unchecked")
28      void fetchesData() throws Exception {
29          final Valve valve = new QueryValve();
30          final Credentials credentials = Mockito.mock(Credentials.class);
31          final ImmutableMap<String, AttributeValue> item =
32              new ImmutableMap.Builder<String, AttributeValue>()
33                  .build();
34          final DynamoDbClient aws = Mockito.mock(DynamoDbClient.class);
35          Mockito.doReturn(aws).when(credentials).aws();
36          Mockito.doReturn(
37              QueryResponse.builder()
38                  .items(
39                      Collections.singletonList(item)
40                  )
41                  .consumedCapacity(
42                      ConsumedCapacity.builder().capacityUnits(1.0d).build()
43                  )
44                  .build()
45          ).when(aws).query(Mockito.any(QueryRequest.class));
46          final Dosage dosage = valve.fetch(
47              credentials, "table",
48              new Conditions(), new ArrayList<>(0)
49          );
50          MatcherAssert.assertThat("should be false", dosage.hasNext(), Matchers.is(false));
51          MatcherAssert.assertThat("should has item", dosage.items(), Matchers.hasItem(item));
52      }
53  
54  }
