5   package com.jcabi.dynamo;
6   
7   import com.jcabi.immutable.ArrayMap;
8   import java.util.Collections;
9   import java.util.Map;
10  import org.hamcrest.MatcherAssert;
11  import org.hamcrest.Matchers;
12  import org.junit.jupiter.api.Test;
13  import software.amazon.awssdk.services.dynamodb.model.AttributeValue;
14  import software.amazon.awssdk.services.dynamodb.model.ExpectedAttributeValue;
15  
16  /**
17   * Test case for {@link Attributes}.
18   * @since 0.1
19   */
20  @SuppressWarnings("PMD.UseConcurrentHashMap")
21  final class AttributesTest {
22  
23      @Test
24      void workAsMapOfAttributes() {
25          final String attr = "id";
26          final AttributeValue value = AttributeValue.builder().s("some text value").build();
27          final Map<String, AttributeValue> attrs = new Attributes()
28              .with(attr, value);
29          MatcherAssert.assertThat("should has size 1", attrs.keySet(), Matchers.hasSize(1));
30          MatcherAssert.assertThat("should has some entry", attrs, Matchers.hasEntry(attr, value));
31          MatcherAssert.assertThat(
32              "should has some text value",
33              new Attributes(attrs),
34              Matchers.hasEntry(attr, value)
35          );
36      }
37  
38      @Test
39      void buildsExpectedKeys() {
40          final String attr = "attr-13";
41          final String value = "some value \u20ac";
42          MatcherAssert.assertThat(
43              "should has 'some value \u20ac'",
44              new Attributes().with(attr, value).asKeys(),
45              Matchers.hasEntry(
46                  attr,
47                  ExpectedAttributeValue.builder().value(
48                      AttributeValue.builder().s(value).build()
49                  ).build()
50              )
51          );
52      }
53  
54      @Test
55      void filtersOutUnnecessaryKeys() {
56          MatcherAssert.assertThat(
57              "should be empty match",
58              new Attributes()
59                  .with("first", "test-1")
60                  .with("second", "test-2")
61                  .only(Collections.singletonList("never"))
62                  .keySet(),
63              Matchers.empty()
64          );
65      }
66  
67      @Test
68      void caseSensitive() {
69          final String first = "Alpha";
70          final String second = "AlPha";
71          final String third = "Beta";
72          MatcherAssert.assertThat(
73              "should has size 2",
74              new Attributes().with(
75                  new ArrayMap<String, AttributeValue>()
76                      .with("Gamma", AttributeValue.builder().s("").build())
77                      .with("gAMma", AttributeValue.builder().s("").build())
78              ).keySet(),
79              Matchers.hasSize(2)
80          );
81          MatcherAssert.assertThat(
82              "should has keys 'Alpha', 'AlPha'",
83              new Attributes()
84                  .with(first, "val-1")
85                  .with(second, "val-2"),
86              Matchers.allOf(
87                  Matchers.hasKey(first),
88                  Matchers.hasKey(second)
89              )
90          );
91          MatcherAssert.assertThat(
92              "should has key 'Beta'",
93              new Attributes()
94                  .with(third, "some text to use")
95                  .only(Collections.singletonList(third)),
96              Matchers.hasKey(third)
97          );
98      }
99  
100 }
