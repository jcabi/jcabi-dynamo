42  /**
43   * Test case for {@link AttributesUpdates}.
44   * @since 0.22
45   */
46  @SuppressWarnings("PMD.TooManyMethods")
47  public final class AttributeUpdatesTest {
48  
49      /**
50       * AttributesUpdates can tell if it's empty or not.
51       */
52      @Test
53      public void tellsIfEmptyOrNot() {
54          final AttributeUpdates attr = new AttributeUpdates();
55          MatcherAssert.assertThat(attr.isEmpty(), Matchers.is(Boolean.TRUE));
56          MatcherAssert.assertThat(
57              attr.with("testkey", Mockito.mock(AttributeValueUpdate.class))
58                  .isEmpty(),
59              Matchers.is(Boolean.FALSE)
60          );
61      }
62  
63      /**
64       * AttributesUpdates can add a new AttributeValueUpdate.
65       */
66      @Test
67      public void addsAttributeValueUpdate() {
68          MatcherAssert.assertThat(0, Matchers.is(0));
69          final AttributeUpdates attr = new AttributeUpdates();
70          MatcherAssert.assertThat(
71              attr.with("testkey1", Mockito.mock(AttributeValueUpdate.class))
72                  .size(),
73              Matchers.is(1)
74          );
75      }
76  
77      /**
78       * AttributesUpdates can add a new AttributeValue.
79       */
80      @Test
81      public void addsAttributeValue() {
82          MatcherAssert.assertThat(0, Matchers.is(0));
83          final AttributeUpdates attr = new AttributeUpdates();
84          MatcherAssert.assertThat(
85              attr.with("testkey2", Mockito.mock(AttributeValue.class)).size(),
86              Matchers.is(1)
87          );
88      }

90      /**
91       * AttributesUpdates can add a new Object.
92       */
93      @Test

98              attr.with("testkey3", "value here").size(),
99              Matchers.is(1)
100         );
101     }

103     /**
104      * AttributesUpdates can return its key Set.
105      */
106     @Test
107     public void returnsKeySet() {
108         final String firstkey = "key1";
109         final AttributeUpdates attr = new AttributeUpdates()
110             .with(firstkey, "valuediff").with("key2", "value2");
111         MatcherAssert.assertThat(
112             attr.keySet().size(),
113             Matchers.is(2)
114         );
115         MatcherAssert.assertThat(
116             attr.keySet().iterator().next(),
117             Matchers.equalTo(firstkey)
118         );
119     }
120 
121     /**
122      * AttributesUpdates can return its values.
123      */
124     @Test
125     public void returnsValues() {
126         final String firstvalue = "value3";
127         final AttributeUpdates attr = new AttributeUpdates()
128             .with("key3", firstvalue).with("key4", "value4");
129         MatcherAssert.assertThat(
130             attr.values().size(),
131             Matchers.is(2)
132         );
133         MatcherAssert.assertThat(
134             attr.values().iterator().next().getValue().getS(),
135             Matchers.equalTo(firstvalue)
136         );
137     }

166     /**

168      */
169     @Test
170     public void containsKey() {
171         final String key = "key11";
172         final AttributeUpdates attr = new AttributeUpdates()
173             .with(key, "value11").with("otherkey1", "othervalue1");
174         MatcherAssert.assertThat(
175             attr.containsKey(key),
176             Matchers.is(Boolean.TRUE)
177         );
178     }
179 
180     /**
181      * AttributesUpdates can tell if it contains a value.
182      */
183     @Test
184     public void containsValue() {
185         final String value = "attrv";
186         final AttributeUpdates attr = new AttributeUpdates()
187             .with("attrkey", value).with("otherkey", "othervalue");
188         MatcherAssert.assertThat(
189             attr.containsValue(
190                 new AttributeValueUpdate(
191                     new AttributeValue(value), AttributeAction.PUT
192                 )
193             ),
194             Matchers.is(Boolean.TRUE)
195         );
196     }
197 
198     /**
199      * AttributesUpdates can return a string representation of itself.
200      */
201     @Test
202     public void canTurnToString() {
203         final AttributeUpdates attr = new AttributeUpdates()
204             .with("onekey", "onevalue").with("secondkey", "secondvalue");
205         MatcherAssert.assertThat(
206             attr.toString(),
207             Matchers.equalTo(
208                 StringUtils.join(
209                     "onekey={Value: {S: onevalue,},Action: PUT}; ",
210                     "secondkey={Value: {S: secondvalue,},Action: PUT}"
211                 )
212             )
213         );
214     }

225                 .size(),
