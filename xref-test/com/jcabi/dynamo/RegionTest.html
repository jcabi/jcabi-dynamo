14  final class RegionTest {

16      @Test
17      void appendsPrefixesToTableNames() {
18          final Table table = Mockito.mock(Table.class);
19          final Region region = Mockito.mock(Region.class);
20          Mockito.doReturn(table).when(region).table(Mockito.anyString());

22          Mockito.verify(region).table("foo-test");
23      }
