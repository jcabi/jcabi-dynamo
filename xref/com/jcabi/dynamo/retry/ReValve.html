5   package com.jcabi.dynamo.retry;
6   
7   import com.jcabi.aspects.Immutable;
8   import com.jcabi.aspects.Loggable;
9   import com.jcabi.aspects.RetryOnFailure;
10  import com.jcabi.dynamo.Credentials;
11  import com.jcabi.dynamo.Dosage;
12  import com.jcabi.dynamo.Valve;
13  import java.io.IOException;
14  import java.util.Collection;
15  import java.util.Map;
16  import java.util.concurrent.TimeUnit;
17  import lombok.EqualsAndHashCode;
18  import lombok.ToString;
19  import software.amazon.awssdk.services.dynamodb.model.Condition;
20  
21  /**
22   * Valve that retries on failure.
23   *
24   * @since 0.9
25   */
26  @Immutable
27  @Loggable(Loggable.DEBUG)
28  @ToString
29  @EqualsAndHashCode(of = "origin")
30  public final class ReValve implements Valve {
31  
32      /**
33       * Original valve.
34       */
35      private final transient Valve origin;
36  
37      /**
38       * Public ctor.
39       * @param valve Origin valve
40       */
41      public ReValve(final Valve valve) {
42          this.origin = valve;
43      }
44  
45      // @checkstyle ParameterNumber (7 lines)
46      @Override
47      @RetryOnFailure(verbose = false, delay = 5, unit = TimeUnit.SECONDS)
48      public Dosage fetch(final Credentials credentials, final String table,
49          final Map<String, Condition> conditions,
50          final Collection<String> keys) throws IOException {
51          return new ReDosage(
52              this.origin.fetch(credentials, table, conditions, keys)
53          );
54      }
55  
56      @Override
57      @RetryOnFailure(verbose = false, delay = 5, unit = TimeUnit.SECONDS)
58      public int count(final Credentials credentials, final String table,
59          final Map<String, Condition> conditions) throws IOException {
60          return this.origin.count(credentials, table, conditions);
61      }
62  
63  }
