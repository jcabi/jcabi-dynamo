30  package com.jcabi.dynamo;
31  
32  import com.amazonaws.services.dynamodbv2.AmazonDynamoDB;
33  import com.jcabi.aspects.Immutable;
34  import com.jcabi.aspects.Loggable;
35  import lombok.EqualsAndHashCode;
36  import lombok.ToString;
37  
38  /**
39   * Amazon DynamoDB region.
40   *
41   * <p>It is recommended to use {@link Region.Simple} in most cases.
42   *
43   * <p>You can use {@link #aws()} method to get access to Amazon DynamoDB
44   * client directly.
45   *
46   * <p>Since version 0.9 it is strongly recommended to wrap your region
47   * in {@link com.jcabi.dynamo.retry.ReRegion} before use, for example:
48   *
49   * <pre> Region region = new ReRegion(new Region.Simple(credentials));</pre>
50   *
51   * <p>After all operations with the region are finished, it can be optionally
52   * shutdown invoking {@link AmazonDynamoDB#shutdown()}. Callers are not expected
53   * to call it, but can if they want to explicitly release any open resources and
54   * forcibly terminate all pending asynchronous service calls. Once a client has
55   * been shutdown, it should not be used to make any more requests.
56   *
57   * @since 0.1
58   */
59  @Immutable
60  public interface Region {
61  
62      /**
63       * Get DynamoDB client.
64       * @return The client
65       */
66      AmazonDynamoDB aws();
67  
68      /**
69       * Get one table.
70       * @param name Table name
71       * @return Table
72       */
73      Table table(String name);
74  
75      /**
76       * Simple region, basic implementation.
77       *
78       * @since 0.1
79       */
80      @Immutable
81      @Loggable(Loggable.DEBUG)
82      @ToString
83      @EqualsAndHashCode(of = "credentials")
84      final class Simple implements Region {
85          /**
86           * Credentials.
87           */
88          private final transient Credentials credentials;
89  
90          /**
91           * Public ctor.
92           * @param creds Credentials
93           */
94          public Simple(final Credentials creds) {
95              this.credentials = creds;
96          }
97  
98          @Override
99          public AmazonDynamoDB aws() {
100             return this.credentials.aws();
101         }
102 
103         @Override
104         public Table table(final String name) {
105             return new AwsTable(this.credentials, this, name);
106         }
107     }
108 
109     /**
110      * All tables have a prefix in front of their names.
111      *
112      * <p>The region has to be used in combination with another region,
113      * for example {@link Region.Simple}:
114      *
115      * <pre>Region region = new Region.Prefixed(
116      *   new Region.Simple(creds),
117      *   "foo-"
118      * );</pre>
119      *
120      * <p>Now, {@code region.table("test")} will return a {@link Table}
121      * instance pointing to the Dynamo DB table named {@code "foo-test"}. Could
122      * be a convenient mechanism when you have many tables for different
123      * projects in the same region.
124      *
125      * @since 0.1
126      */
127     @Immutable
128     @Loggable(Loggable.DEBUG)
129     @ToString
130     @EqualsAndHashCode(of = { "origin", "prefix" })
131     final class Prefixed implements Region {
132         /**
133          * Original region.
134          */
135         private final transient Region origin;
136 
137         /**
138          * Prefix to add.
139          */
140         private final transient String prefix;
141 
142         /**
143          * Public ctor.
144          * @param region Original region
145          * @param pfx Prefix to add to all tables
146          */
147         public Prefixed(final Region region, final String pfx) {
148             this.origin = region;
149             this.prefix = pfx;
150         }
151 
152         @Override
153         public AmazonDynamoDB aws() {
154             return this.origin.aws();
155         }
156 
157         @Override
158         public Table table(final String name) {
159             return this.origin.table(
160                 new StringBuilder(this.prefix).append(name).toString()
161             );
162         }
163     }
164 
165 }
