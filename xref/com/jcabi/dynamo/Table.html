5   package com.jcabi.dynamo;
6   
7   import com.jcabi.aspects.Immutable;
8   import java.io.IOException;
9   import java.util.Map;
10  import software.amazon.awssdk.services.dynamodb.model.AttributeValue;
11  
12  /**
13   * Amazon DynamoDB table abstraction.
14   *
15   * <p>To get data from the table use {@link Table#frame()} method. To
16   * create a new item in the table (or replace the existing one) use
17   * {@link Table#put(Attributes)} method. For example:
18   *
19   * <pre> Region region = new Region.Simple(...);
20   * Table table = region.table("employees");
21   * table.put(new Attributes().with("name", "John Smith"));
22   * for (Item item : table.frame()) {
23   *   System.out.println("Name: " + item.get("name").s());
24   * }
25   * table.frame()
26   *   .where("name", Conditions.equalTo("John Smith"))
27   *   .iterator().next().remove();</pre>
28   *
29   * @since 0.1
30   */
31  @Immutable
32  public interface Table {
33  
34      /**
35       * Put new item there.
36       *
37       * <p>It is recommended to use {@link Attributes} supplementary class,
38       * instead of a raw {@link Map}.
39       *
40       * @param attributes Attributes to save
41       * @return Item just created
42       * @throws IOException In case of DynamoDB failure
43       * @see Attributes
44       */
45      Item put(Map<String, AttributeValue> attributes) throws IOException;
46  
47      /**
48       * Make a new frame, in order to retrieve items.
49       * @return Frame
50       */
51      Frame frame();
52  
53      /**
54       * Get back to the entire region.
55       * @return Region
56       */
57      Region region();
58  
59      /**
60       * Get real table name.
61       * @return Actual name of DynamoDB table
62       */
63      String name();
64  
65      /**
66       * Delete item from aws table.
67       * <p>It is recommended to use {@link Attributes} supplementary class,
68       * instead of a raw {@link Map}.
69       * @param attributes Attributes containing item key and value
70       * @throws IOException In case of DynamoDB failure
71       * @see Attributes
72       */
73      void delete(Map<String, AttributeValue> attributes) throws IOException;
74  
75  }
