5   package com.jcabi.dynamo;
6   
7   import com.jcabi.aspects.Immutable;
8   import java.io.IOException;
9   import java.util.Collection;
10  import java.util.Map;
11  import software.amazon.awssdk.services.dynamodb.model.Condition;
12  
13  /**
14   * Provider of dosages of DynamoDB items.
15   *
16   * @since 0.7.21
17   */
18  @Immutable
19  public interface Valve {
20  
21      /**
22       * Fetch the first dosage.
23       * @param credentials Credentials to AWS
24       * @param table Table name
25       * @param conditions Conditions
26       * @param keys Keys of the table
27       * @return Dosage
28       * @throws IOException In case of DynamoDB failure
29       * @checkstyle ParameterNumber (5 lines)
30       */
31      Dosage fetch(Credentials credentials, String table,
32          Map<String, Condition> conditions, Collection<String> keys)
33          throws IOException;
34  
35      /**
36       * Count items.
37       * @param credentials Credentials to AWS
38       * @param table Table name
39       * @param conditions Conditions
40       * @return Total count of the
41       * @throws IOException In case of DynamoDB failure
42       */
43      int count(Credentials credentials, String table,
44          Map<String, Condition> conditions) throws IOException;
45  
46  }
