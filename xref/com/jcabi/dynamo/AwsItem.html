16  import lombok.ToString;
17  import software.amazon.awssdk.core.exception.SdkClientException;
18  import software.amazon.awssdk.services.dynamodb.DynamoDbClient;
19  import software.amazon.awssdk.services.dynamodb.model.AttributeValue;
20  import software.amazon.awssdk.services.dynamodb.model.AttributeValueUpdate;

24  import software.amazon.awssdk.services.dynamodb.model.ReturnValue;
25  import software.amazon.awssdk.services.dynamodb.model.UpdateItemRequest;
26  import software.amazon.awssdk.services.dynamodb.model.UpdateItemResponse;

28  /**
29   * Single item/row in a DynamoDB table.
30   *
31   * @since 0.1
32   */
33  @Immutable
34  @Loggable(Loggable.DEBUG)
35  @ToString
36  @EqualsAndHashCode(of = { "credentials", "frm", "name", "attributes" })
37  final class AwsItem implements Item {
38  
39      /**
40       * AWS credentials.
41       */
42      private final transient Credentials credentials;
43  
44      /**
45       * Frame.
46       */
47      private final transient AwsFrame frm;
48  
49      /**
50       * Table name.
51       */
52      private final transient String name;
53  
54      /**
55       * Pre-loaded attributes and their values.
56       */
57      private final transient Attributes attributes;
58  
59      /**
60       * Table keys.
61       */
62      private final transient Array<String> keys;
63  
64      /**
65       * Public ctor.
66       * @param creds Credentials
67       * @param frame Frame
68       * @param table Table name
69       * @param attrs Loaded already attributes with values
70       * @param pks Keys of the table
71       * @checkstyle ParameterNumber (5 lines)
72       */
73      AwsItem(final Credentials creds, final AwsFrame frame,
74          final String table, final Attributes attrs,
75          final Array<String> pks) {
76          this.credentials = creds;
77          this.frm = frame;
78          this.name = table;
79          this.attributes = attrs;
80          this.keys = pks;
81      }
82  
83      @Override
84      public boolean has(final String attr) throws IOException {
85          final String attrib = attr;
86          boolean has = this.attributes.containsKey(attrib);
87          if (!has) {
88              final DynamoDbClient aws = this.credentials.aws();
89              try {
90                  final GetItemResponse result = aws.getItem(
91                      this.makeItemRequestFor(attr)
92                  );
93                  has = result.item().get(attrib) != null;
94                  Logger.info(
95                      this, "#has('%s'): %B from DynamoDB, %s",
96                      attr, has,
97                      new PrintableConsumedCapacity(
98                          result.consumedCapacity()
99                      ).print()
100                 );
101             } catch (final SdkClientException ex) {
102                 throw new IOException(
103                     String.format(
104                         "Failed to check existence of \"%s\" at \"%s\" by %s",
105                         attr, this.name, this.keys
106                     ),
107                     ex
108                 );
109             } finally {
110                 aws.close();
111             }
112         }
113         return has;
114     }
115 
116     @Override
117     public AttributeValue get(final String attr) throws IOException {
118         final String attrib = attr;
119         AttributeValue value = this.attributes.get(attrib);
120         if (value == null) {
121             final DynamoDbClient aws = this.credentials.aws();
122             try {
123                 final GetItemResponse result = aws.getItem(
124                     this.makeItemRequestFor(attrib)
125                 );
126                 value = result.item().get(attrib);
127                 Logger.info(
128                     this,
129                     "#get('%s'): loaded '%[text]s' from DynamoDB, %s",
130                     attrib, value,
131                     new PrintableConsumedCapacity(
132                         result.consumedCapacity()
133                     ).print()
134                 );
135             } catch (final SdkClientException ex) {
136                 throw new IOException(
137                     String.format(
138                         "Failed to get \"%s\" from \"%s\" by %s",
139                         attr, this.name, this.keys
140                     ),
141                     ex
142                 );
143             } finally {
144                 aws.close();
145             }
146         }
147         if (value == null) {
148             throw new NoSuchElementException(
149                 String.format("attribute \"%s\" not found", attr)
150             );
151         }
152         return value;
153     }
154 
155     @Override
156     public Map<String, AttributeValue> put(final String attr,
157         final AttributeValueUpdate value) throws IOException {
158         return this.put(new AttributeUpdates().with(attr, value));
159     }
160 
161     @Override
162     public Map<String, AttributeValue> put(
163         final Map<String, AttributeValueUpdate> attrs) throws IOException {
164         final DynamoDbClient aws = this.credentials.aws();
165         final Attributes expected = this.attributes.only(this.keys);
166         try {
167             final UpdateItemResponse result = aws.updateItem(
168                 UpdateItemRequest.builder()
169                     .tableName(this.name)
170                     .expected(expected.asKeys())
171                     .key(expected)
172                     .attributeUpdates(attrs)
173                     .returnConsumedCapacity(ReturnConsumedCapacity.TOTAL)
174                     .returnValues(ReturnValue.UPDATED_NEW)
175                     .build()
176             );
177             Logger.info(
178                 this, "#put('%s'): updated item to DynamoDB, %s",
179                 attrs,
180                 new PrintableConsumedCapacity(
181                     result.consumedCapacity()
182                 ).print()
183             );
184             return result.attributes();
185         } catch (final SdkClientException ex) {
186             throw new IOException(
187                 String.format(
188                     "Failed to put %s into \"%s\" with %s",
189                     attrs, this.name, this.keys
190                 ),
191                 ex
192             );
193         } finally {
194             aws.close();
195         }
196     }
197 
198     @Override
199     public Frame frame() {
200         return this.frm;
201     }
202 
203     /**
204      * Makes a GetItemRequest for a given attribute.
205      * @param attr Attribute name
206      * @return GetItemRequest
207      */
208     private GetItemRequest makeItemRequestFor(final String attr) {
209         return GetItemRequest.builder()
210             .tableName(this.name)
211             .attributesToGet(Collections.singletonList(attr))
212             .key(this.attributes.only(this.keys))
213             .returnConsumedCapacity(ReturnConsumedCapacity.TOTAL)
214             .consistentRead(true)
215             .build();
216     }
217 
218 }
