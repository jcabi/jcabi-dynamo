5   package com.jcabi.dynamo;
6   
7   import software.amazon.awssdk.services.dynamodb.model.ProvisionedThroughput;
8   import software.amazon.awssdk.services.dynamodb.model.UpdateTableRequest;
9   
10  /**
11   * Throughput of a table.
12   * @since 0.18.4
13   */
14  public final class Throughput {
15      /**
16       * Table.
17       */
18      private final transient Table table;
19  
20      /**
21       * Public ctor.
22       * @param tbl The table.
23       */
24      public Throughput(final Table tbl) {
25          this.table = tbl;
26      }
27  
28      /**
29       * Adjusts throughput on the table.
30       */
31      public void adjust() {
32          this.table
33              .region()
34              .aws()
35              .updateTable(
36                  UpdateTableRequest.builder()
37                      .tableName(this.table.name())
38                      .provisionedThroughput(
39                          Throughput.suitableThroughput()
40                      )
41                      .build()
42              );
43      }
44  
45      /**
46       * Decides which throughput value is the most suitable according to
47       * certain parameters of elasticity/scalability.
48       * @return Throughput settings.
49       * @todo #10 The exact algorithm for figuring out read and write
50       *  capacities should be based on a CloudWatch metric accessed with
51       *  credentials of this.table.region().aws().
52       */
53      private static ProvisionedThroughput suitableThroughput() {
54          return ProvisionedThroughput.builder()
55              .readCapacityUnits((long) 100)
56              .writeCapacityUnits((long) 100)
57              .build();
58      }
59  }
