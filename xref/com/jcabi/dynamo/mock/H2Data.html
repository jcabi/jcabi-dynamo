82      /**
83       * Fetcher of rows.
84       */
85      private static final Outcome<Iterable<Attributes>> OUTCOME =
86          // @checkstyle AnonInnerLengthCheck (50 lines)
87          new Outcome<Iterable<Attributes>>() {
88              @Override
89              public Iterable<Attributes> handle(final ResultSet rset,
90                  final Statement stmt) throws SQLException {
91                  final Collection<Attributes> items =
92                      new LinkedList<Attributes>();
93                  while (rset.next()) {
94                      items.add(this.fetch(rset));
95                  }
96                  return items;
97              }
98              /**
99               * Convert result set to Attributes.
100              * @param rset Result set
101              * @return Attribs
102              * @throws SQLException If fails
103              */
104             private Attributes fetch(final ResultSet rset) throws SQLException {
105                 final ResultSetMetaData meta = rset.getMetaData();
106                 Attributes attrs = new Attributes();
107                 for (int idx = 0; idx < meta.getColumnCount(); ++idx) {
108                     final String text = rset.getString(idx + 1);
109                     AttributeValue value = new AttributeValue().withS(text);
110                     if (text.matches("[0-9]+")) {
111                         value = value.withN(text);
112                     }
113                     attrs = attrs.with(
114                         meta.getColumnName(idx + 1).toLowerCase(Locale.ENGLISH),
115                         value
116                     );
117                 }
118                 return attrs;
119             }
120         };

184     @Override
185     public Iterable<String> keys(final String table) throws IOException {
186         try {
187             return Iterables.transform(
188                 new JdbcSession(this.connection())
189                     // @checkstyle LineLength (1 line)
190                     .sql("SELECT SQL FROM INFORMATION_SCHEMA.CONSTRAINTS WHERE TABLE_NAME = ?")
191                     .set(H2Data.encodeTableName(table))
192                     .select(
193                         new ListOutcome<String>(
194                             new ListOutcome.Mapping<String>() {
195                                 @Override
196                                 public String map(final ResultSet rset)
197                                     throws SQLException {
198                                     return rset.getString(1);
199                                 }
200                             }
201                         )
202                     ),
203                 new Function<String, String>() {
204                     @Override
205                     public String apply(final String input) {
206                         final Matcher matcher = Pattern.compile(
207                             "PRIMARY KEY\\((.*)\\)"
208                         ).matcher(input);
209                         if (!matcher.find()) {
210                             throw new IllegalStateException(
211                                 String.format(
212                                     "something is wrong here: \"%s\"", input
213                                 )
214                             );
215                         }
216                         return matcher.group(1).toLowerCase(Locale.ENGLISH);
217                     }
218                 }
219             );
220         } catch (final SQLException ex) {
221             throw new IOException(ex);
222         }
223     }

225     @Override
226     public Iterable<Attributes> iterate(final String table,
227         final Conditions conds) throws IOException {
228         try {
229             final StringBuilder sql = new StringBuilder("SELECT * FROM ")
230                 .append(H2Data.encodeTableName(table));
231             if (!conds.isEmpty()) {
232                 sql.append(" WHERE ");
233                 Joiner.on(H2Data.AND).appendTo(
234                     sql,
235                     Iterables.transform(conds.keySet(), H2Data.WHERE)
236                 );
237             }
238             JdbcSession session = new JdbcSession(this.connection())
239                 .sql(sql.toString());
240             for (final Condition cond : conds.values()) {
241                 if (cond.getAttributeValueList().size() != 1) {
242                     throw new UnsupportedOperationException(
243                         "at the moment only one value of condition is supported"
244                     );
245                 }
246                 if (!cond.getComparisonOperator()
247                     .equals(ComparisonOperator.EQ.toString())) {
248                     throw new UnsupportedOperationException(
249                         String.format(
250                             "at the moment only EQ operator is supported: %s",
251                             cond.getComparisonOperator()
252                         )
253                     );
254                 }
255                 final AttributeValue val = cond.getAttributeValueList().get(0);
256                 session = session.set(H2Data.value(val));
257             }
258             return session.select(H2Data.OUTCOME);
259         } catch (final SQLException ex) {
260             throw new IOException(ex);
261         }
262     }

264     @Override
265     public void put(final String table, final Attributes attrs)
266         throws IOException {
267         try {
268             JdbcSession session = new JdbcSession(this.connection());
269             for (final AttributeValue value : attrs.values()) {
270                 session = session.set(H2Data.value(value));
271             }
272             session.sql(
273                 String.format(
274                     "INSERT INTO %s (%s) VALUES (%s)",
275                     H2Data.encodeTableName(table),
276                     Joiner.on(',').join(attrs.keySet()),
277                     Joiner.on(',').join(Collections.nCopies(attrs.size(), "?"))
278                 )
279             );
280             session.execute();
281         } catch (final SQLException ex) {
282             throw new IOException(ex);
283         }
284     }

286     @Override
287     public void update(final String table, final Attributes keys,
288         final AttributeUpdates attrs)
289         throws IOException {
290         try {
291             JdbcSession session = new JdbcSession(this.connection());
292             for (final AttributeValueUpdate value : attrs.values()) {
293                 session = session.set(H2Data.value(value.getValue()));
294             }
295             for (final AttributeValue value : keys.values()) {
296                 session = session.set(H2Data.value(value));
297             }
298             session.sql(
299                 String.format(
300                     "UPDATE %s SET %s WHERE %s",
301                     H2Data.encodeTableName(table),
302                     Joiner.on(',').join(
303                         Iterables.transform(attrs.keySet(), H2Data.WHERE)
304                     ),
305                     Joiner.on(H2Data.AND).join(
306                         Iterables.transform(keys.keySet(), H2Data.WHERE)
307                     )
308                 )
309             );
310             session.execute();
311         } catch (final SQLException ex) {
312             throw new IOException(ex);
313         }
314     }

316     @Override
317     public void delete(final String table, final Attributes keys)
318         throws IOException {
319         try {
320             JdbcSession session = new JdbcSession(this.connection());
321             for (final AttributeValue value : keys.values()) {
322                 session = session.set(H2Data.value(value));
323             }
324             session.sql(
325                 String.format(
326                     "DELETE FROM %s WHERE %s",
327                     H2Data.encodeTableName(table),
328                     Joiner.on(H2Data.AND).join(
329                         Iterables.transform(keys.keySet(), H2Data.WHERE)
330                     )
331                 )
332             );
333             session.execute();
334         } catch (final SQLException ex) {
335             throw new IOException(ex);
336         }
337     }

339     /**
340      * With this table, that has given primary keys.
341      * @param table Table name
342      * @param keys Primary keys
343      * @param attrs Attributes
344      * @return New data, modified
345      * @throws IOException If fails
346      */
347     public H2Data with(final String table, final String[] keys,
348         final String... attrs) throws IOException {
349         if (keys.length == 0) {
350             throw new IllegalArgumentException(
351                 String.format("empty list of keys for %s table", table)
352             );
353         }
354         final StringBuilder sql = new StringBuilder("CREATE TABLE ")
355             .append(H2Data.encodeTableName(table)).append(" (");
356         Joiner.on(',').appendTo(
357             sql,
358             Iterables.transform(Arrays.asList(keys), H2Data.CREATE_KEY)
359         );
360         if (attrs.length > 0) {
361             sql.append(',');
362             Joiner.on(',').appendTo(
363                 sql,
364                 Iterables.transform(Arrays.asList(attrs), H2Data.CREATE_ATTR)
365             );
366         }
367         sql.append(')');
368         try {
369             new JdbcSession(this.connection()).sql(sql.toString()).execute();
370         } catch (final SQLException ex) {
371             throw new IOException(ex);
372         }
373         return this;
374     }
375     /**
376      * Make data source.
377      * @return Data source for JDBC
378      * @throws SQLException If fails
379      */
380     private Connection connection() throws SQLException {
381         return new Driver().connect(this.jdbc, new Properties());
382     }

384     /**
385      * Get value from attribute.
386      * @param attr Attribute value
387      * @return Text format
388      */
389     private static String value(final AttributeValue attr) {
390         String val = attr.getS();
391         if (val == null) {
392             val = attr.getN();
393         }
394         if (val == null) {
395             throw new IllegalArgumentException(
396                 "we support only N and S at the moment"
397             );
398         }
399         return val;
400     }

402     /**
403      * Base32-encodes table name for use with H2.
404      * @param table Table name to encode
405      * @return Base-32-encoded table name
406      */
407     private static String encodeTableName(final String table) {
408         return Joiner.on("").join(
409             "_",
410             new Base32(true, (byte) '_').encodeAsString(table.getBytes())
411         );
412     }
