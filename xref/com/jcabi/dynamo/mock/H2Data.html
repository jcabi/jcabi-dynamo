70   */
71  @Immutable
72  @ToString
73  @Loggable(Loggable.DEBUG)
74  @EqualsAndHashCode(of = "jdbc")
75  @SuppressWarnings({ "PMD.TooManyMethods", "PMD.ExcessiveImports" })

78      /**
79       * Fetcher of rows.
80       */
81      private static final Outcome<Iterable<Attributes>> OUTCOME =
82          // @checkstyle AnonInnerLengthCheck (50 lines)
83          new Outcome<Iterable<Attributes>>() {
84              @Override
85              public Iterable<Attributes> handle(final ResultSet rset,
86                  final Statement stmt) throws SQLException {
87                  final Collection<Attributes> items = new LinkedList<>();
88                  while (rset.next()) {
89                      items.add(this.fetch(rset));
90                  }
91                  return items;
92              }
93  
94              /**
95               * Convert result set to Attributes.
96               * @param rset Result set
97               * @return Attribs
98               * @throws SQLException If fails
99               */
100             private Attributes fetch(final ResultSet rset) throws SQLException {
101                 final ResultSetMetaData meta = rset.getMetaData();
102                 Attributes attrs = new Attributes();
103                 for (int idx = 0; idx < meta.getColumnCount(); ++idx) {
104                     final String text = rset.getString(idx + 1);
105                     AttributeValue value = new AttributeValue().withS(text);
106                     if (text.matches("[0-9]+")) {
107                         value = value.withN(text);
108                     }
109                     attrs = attrs.with(
110                         meta.getColumnName(idx + 1).toLowerCase(Locale.ENGLISH),
111                         value
112                     );
113                 }
114                 return attrs;
115             }
116         };

118     /**
119      * Where clause.
120      */
121     private static final Function<String, String> WHERE =
122         key -> String.format("%s = ?", key);
123 
124     /**
125      * Select WHERE.
126      * @checkstyle AnonInnerLengthCheck (100 lines)
127      * @checkstyle LineLength (3 lines)
128      */
129     private static final Function<Map.Entry<String, Condition>, String> SELECT_WHERE = cnd -> {
130         final String opr;
131         if (cnd.getValue().getComparisonOperator()
132             .equals(ComparisonOperator.GT.toString())) {
133             opr = ">";
134         } else if (cnd.getValue().getComparisonOperator()
135             .equals(ComparisonOperator.LT.toString())) {
136             opr = "<";
137         } else if (cnd.getValue().getComparisonOperator()
138             .equals(ComparisonOperator.EQ.toString())) {
139             opr = "=";
140         } else {
141             throw new UnsupportedOperationException(
142                 String.format(
143                     // @checkstyle LineLength (1 line)
144                     "At the moment only EQ/GT/LT operators are supported: %s",
145                     cnd.getValue().getComparisonOperator()
146                 )
147             );
148         }
149         return String.format(
150             "%s %s ?", cnd.getKey(), opr
151         );
152     };

195     @Override
196     public Iterable<String> keys(final String table) throws IOException {
197         try {
198             return new JdbcSession(this.connection())
199                 // @checkstyle LineLength (1 line)
200                 .sql("SELECT COLUMN_NAME FROM INFORMATION_SCHEMA.KEY_COLUMN_USAGE WHERE TABLE_NAME = ?")
201                 .set(H2Data.encodeTableName(table))
202                 .select(
203                     new ListOutcome<>(
204                         rset -> rset.getString(1).toLowerCase(Locale.ENGLISH)
205                     )
206                 );
207         } catch (final SQLException ex) {
208             throw new IOException(ex);
209         }
210     }
211 
212     @Override
213     public Iterable<Attributes> iterate(final String table,
214         final Conditions conds) throws IOException {
215         try {
216             final StringBuilder sql = new StringBuilder("SELECT * FROM ")
217                 .append(H2Data.encodeTableName(table));
218             if (!conds.isEmpty()) {
219                 sql.append(" WHERE ");
220                 Joiner.on(H2Data.AND).appendTo(
221                     sql,
222                     Iterables.transform(conds.entrySet(), H2Data.SELECT_WHERE)
223                 );
224             }
225             JdbcSession session = new JdbcSession(this.connection())
226                 .sql(sql.toString());
227             for (final Condition cond : conds.values()) {
228                 if (cond.getAttributeValueList().size() != 1) {
229                     throw new UnsupportedOperationException(
230                         "At the moment only one value of condition is supported"
231                     );
232                 }
233                 final AttributeValue val = cond.getAttributeValueList().get(0);
234                 session = session.set(H2Data.value(val));
235             }
236             return session.select(H2Data.OUTCOME);
237         } catch (final SQLException ex) {
238             throw new IOException(ex);
239         }
240     }
241 
242     @Override
243     public void put(final String table, final Attributes attrs)
244         throws IOException {
245         try {
246             JdbcSession session = new JdbcSession(this.connection());
247             for (final AttributeValue value : attrs.values()) {
248                 session = session.set(H2Data.value(value));
249             }
250             session.sql(
251                 String.format(
252                     "INSERT INTO %s (%s) VALUES (%s)",
253                     H2Data.encodeTableName(table),
254                     Joiner.on(',').join(attrs.keySet()),
255                     Joiner.on(',').join(Collections.nCopies(attrs.size(), "?"))
256                 )
257             );
258             session.execute();
259         } catch (final SQLException ex) {
260             throw new IOException(ex);
261         }
262     }
263 
264     @Override
265     public void update(final String table, final Attributes keys,
266         final AttributeUpdates attrs)
267         throws IOException {
268         try {
269             JdbcSession session = new JdbcSession(this.connection());
270             for (final AttributeValueUpdate value : attrs.values()) {
271                 session = session.set(H2Data.value(value.getValue()));
272             }
273             for (final AttributeValue value : keys.values()) {
274                 session = session.set(H2Data.value(value));
275             }
276             session.sql(
277                 String.format(
278                     "UPDATE %s SET %s WHERE %s",
279                     H2Data.encodeTableName(table),
280                     Joiner.on(',').join(
281                         Iterables.transform(attrs.keySet(), H2Data.WHERE)
282                     ),
283                     Joiner.on(H2Data.AND).join(
284                         Iterables.transform(keys.keySet(), H2Data.WHERE)
285                     )
286                 )
287             );
288             session.execute();
289         } catch (final SQLException ex) {
290             throw new IOException(ex);
291         }
292     }
293 
294     @Override
295     public void delete(final String table, final Attributes keys)
296         throws IOException {
297         try {
298             JdbcSession session = new JdbcSession(this.connection());
299             for (final AttributeValue value : keys.values()) {
300                 session = session.set(H2Data.value(value));
301             }
302             session.sql(
303                 String.format(
304                     "DELETE FROM %s WHERE %s",
305                     H2Data.encodeTableName(table),
306                     Joiner.on(H2Data.AND).join(
307                         Iterables.transform(keys.keySet(), H2Data.WHERE)
308                     )
309                 )
310             );
311             session.execute();
312         } catch (final SQLException ex) {
313             throw new IOException(ex);
314         }
315     }

317     /**
318      * With this table, that has given primary keys.
319      * @param table Table name
320      * @param keys Primary keys
321      * @param attrs Attributes
322      * @return New data, modified
323      * @throws IOException If fails
324      */
325     public H2Data with(final String table, final String[] keys,
326         final String... attrs) throws IOException {
327         if (keys.length == 0) {
328             throw new IllegalArgumentException(
329                 String.format("Empty list of keys for %s table", table)
330             );
331         }
332         final StringBuilder sql = new StringBuilder("CREATE TABLE ")
333             .append(H2Data.encodeTableName(table)).append(" (");
334         Joiner.on(',').appendTo(
335             sql,
336             Iterables.transform(Arrays.asList(keys), H2Data.CREATE_KEY)
337         );
338         if (attrs.length > 0) {
339             sql.append(',');
340             Joiner.on(',').appendTo(
341                 sql,
342                 Iterables.transform(Arrays.asList(attrs), H2Data.CREATE_ATTR)
343             );
344         }
345         sql.append(')');
346         try {
347             new JdbcSession(this.connection()).sql(sql.toString()).execute();
348         } catch (final SQLException ex) {
349             throw new IOException(ex);
350         }
351         return this;
352     }

354     /**
355      * Make data source.
356      * @return Data source for JDBC
357      */
358     private DataSource connection() {
359         final JdbcDataSource src = new JdbcDataSource();
360         src.setURL(this.jdbc);
361         return src;
362     }
363 
364     /**
365      * Get value from attribute.
366      * @param attr Attribute value
367      * @return Text format
368      */
369     private static String value(final AttributeValue attr) {
370         String val = attr.getS();
371         if (val == null) {
372             val = attr.getN();
373         }
374         if (val == null) {
375             throw new IllegalArgumentException(
376                 "We support only N and S at the moment"
377             );
378         }
379         return val;
380     }
381 
382     /**
383      * Base32-encodes table name for use with H2.
384      * @param table Table name to encode
385      * @return Base-32-encoded table name
386      */
387     private static String encodeTableName(final String table) {
388         return Joiner.on("").join(
389             "_",
390             new Base32(true, (byte) '_').encodeAsString(table.getBytes())
391         );
392     }
