30  package com.jcabi.dynamo.mock;
31  
32  import com.amazonaws.services.dynamodbv2.model.Condition;
33  import com.google.common.base.Function;
34  import com.google.common.collect.Iterators;
35  import com.jcabi.aspects.Immutable;
36  import com.jcabi.aspects.Loggable;
37  import com.jcabi.dynamo.Attributes;
38  import com.jcabi.dynamo.Conditions;
39  import com.jcabi.dynamo.Frame;
40  import com.jcabi.dynamo.Item;
41  import com.jcabi.dynamo.Table;
42  import com.jcabi.dynamo.Valve;
43  import com.jcabi.immutable.ArrayMap;
44  import java.io.IOException;
45  import java.util.AbstractCollection;
46  import java.util.Iterator;
47  import java.util.Map;
48  import lombok.EqualsAndHashCode;
49  import lombok.ToString;
50  
51  /**
52   * Mock version of {@link Frame}.
53   *
54   * @since 0.10
55   */
56  @Immutable
57  @ToString
58  @Loggable(Loggable.DEBUG)
59  @EqualsAndHashCode(callSuper = false, of = { "tbl", "data", "conds" })
60  final class MkFrame extends AbstractCollection<Item> implements Frame {
61  
62      /**
63       * Data.
64       */
65      private final transient MkData data;
66  
67      /**
68       * Table.
69       */
70      private final transient String tbl;
71  
72      /**
73       * Conditions.
74       */
75      private final transient Conditions conds;
76  
77      /**
78       * Public ctor.
79       * @param dta Data
80       * @param table Table
81       */
82      MkFrame(final MkData dta, final String table) {
83          this(dta, table, new Conditions());
84      }
85  
86      /**
87       * Public ctor.
88       * @param dta Data
89       * @param table Table
90       * @param conditions Map of conditions
91       */
92      MkFrame(final MkData dta, final String table, final Conditions conditions) {
93          super();
94          this.data = dta;
95          this.tbl = table;
96          this.conds = conditions;
97      }
98  
99      @Override
100     public Iterator<Item> iterator() {
101         try {
102             return Iterators.transform(
103                 this.data.iterate(this.tbl, this.conds).iterator(),
104                 new Function<Attributes, Item>() {
105                     @Override
106                     public Item apply(final Attributes input) {
107                         return new MkItem(
108                             MkFrame.this.data,
109                             MkFrame.this.tbl,
110                             input
111                         );
112                     }
113                 }
114             );
115         } catch (final IOException ex) {
116             throw new IllegalStateException(ex);
117         }
118     }
119 
120     @Override
121     public int size() {
122         return Iterators.size(this.iterator());
123     }
124 
125     @Override
126     public Frame where(final String name, final String value) {
127         return this.where(name, Conditions.equalTo(value));
128     }
129 
130     @Override
131     public Frame where(final String name, final Condition condition) {
132         return this.where(
133             new ArrayMap<String, Condition>().with(name, condition)
134         );
135     }
136 
137     @Override
138     public Frame where(final Map<String, Condition> conditions) {
139         return new MkFrame(this.data, this.tbl, this.conds.with(conditions));
140     }
141 
142     @Override
143     public Table table() {
144         return new MkTable(this.data, this.tbl);
145     }
146 
147     @Override
148     public Frame through(final Valve valve) {
149         return this;
150     }
151 }
