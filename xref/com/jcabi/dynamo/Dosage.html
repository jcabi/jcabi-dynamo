5   package com.jcabi.dynamo;
6   
7   import com.jcabi.aspects.Immutable;
8   import com.jcabi.aspects.Loggable;
9   import java.util.ArrayList;
10  import java.util.List;
11  import java.util.Map;
12  import lombok.EqualsAndHashCode;
13  import lombok.ToString;
14  import software.amazon.awssdk.services.dynamodb.model.AttributeValue;
15  
16  /**
17   * Dosage of items retrieved from table.
18   *
19   * @since 0.1
20   */
21  @Immutable
22  public interface Dosage {
23  
24      /**
25       * Items.
26       * @return List of items
27       */
28      List<Map<String, AttributeValue>> items();
29  
30      /**
31       * Has next dosage?
32       * @return TRUE if next storage is available
33       */
34      boolean hasNext();
35  
36      /**
37       * Fetch next dosage.
38       * @return The dosage
39       */
40      Dosage next();
41  
42      /**
43       * Always empty.
44       *
45       * @since 0.1
46       */
47      @Immutable
48      @Loggable(Loggable.DEBUG)
49      @ToString
50      @EqualsAndHashCode
51      final class Empty implements Dosage {
52          @Override
53          public List<Map<String, AttributeValue>> items() {
54              return new ArrayList<>(0);
55          }
56  
57          @Override
58          public Dosage next() {
59              throw new IllegalStateException(
60                  "this is nothing left"
61              );
62          }
63  
64          @Override
65          public boolean hasNext() {
66              return false;
67          }
68      }
69  
70  }
